In [11]:
def input_age(cols):
    Age=cols[0]
    Pclass=cols[1]
    if pd.isnull(Age):
        if Pclass==1:
            return 37
        elif Pclass==2:
            return 29
        else: 
            return 25
    else:
        return Age
In [12]:
data['Age']= data[['Age','Pclass']].apply(input_age,axis=1)
